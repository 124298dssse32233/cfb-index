```python
# src/cfb_rankings/wire/offseason_fallback.py — full rewrite

"""Offseason wire fallback. Surfaces REAL historical content when live
flow is thin. Never fabricates transactions."""

from datetime import date, timedelta
from typing import Iterable
import sqlite3

def select_retro_wire_rows(today: date, db: sqlite3.Connection,
                           n: int = 6) -> Iterable[dict]:
    """Pull real wire rows from same calendar window N years back.

    Priority order:
      1. Same MM-DD ±2 days in 2024, 2023, 2022, 2021, 2020 from wire_entries
         tagged with retro_year
      2. Top recruiting commits from current year (player_recruiting_profiles)
         where star >= 4 and committed in last 14d — real names, real outcomes
      3. Top archive_threads (Arctic Shift) same calendar week N years back

    Every returned row carries:
      - source_kind='retro' (NOT 'unverified')
      - retro_year (renderer prefixes "5 years ago today:")
      - real player names + real outcomes
    """
    target_md = today.strftime("%m-%d")
    out = []

    # Priority 1: Same-MM-DD wire entries from 2020-2024
    for year_offset in range(1, 6):
        rows = db.execute("""
            SELECT *, ? as retro_year FROM wire_entries
            WHERE strftime('%m-%d', occurred_at) = ?
              AND strftime('%Y', occurred_at) = ?
              AND source_kind != 'retro'
              AND source_kind != 'unverified'
            ORDER BY impact_score DESC LIMIT 3
        """, (today.year - year_offset, target_md,
              str(today.year - year_offset))).fetchall()
        out.extend(dict(r) for r in rows)
        if len(out) >= n:
            break

    # Priority 2: Real recent recruiting commits if still under target
    if len(out) < n:
        commits = db.execute("""
            SELECT * FROM player_recruiting_profiles
            WHERE committed_team IS NOT NULL
              AND date(committed_at) >= date(?, '-14 days')
              AND star_rating >= 4
            ORDER BY star_rating DESC, committed_at DESC LIMIT ?
        """, (today.isoformat(), n - len(out))).fetchall()
        out.extend(_recruit_to_wire_dict(dict(c)) for c in commits)

    # Priority 3: Archive threads if still under target
    if len(out) < n:
        threads = db.execute("""
            SELECT * FROM archive_threads
            WHERE strftime('%m-%d', created_at) = ?
              AND score >= 50
            ORDER BY score DESC LIMIT ?
        """, (target_md, n - len(out))).fetchall()
        out.extend(_thread_to_wire_dict(dict(t)) for t in threads)

    return out[:n]
```

3. **Factuality critic** (Sonnet) — every numeric/dated claim traceable to source observations
4. **Engagement critic** (Opus, Pattern D only) — "would a sophisticated CFB reader linger?"
5. **Continuity critic** (Sonnet, Pattern E only) — advances thread, no contradictions, named-entity consistency

- **Rung 2:** consecutive critic failures ≥ 3 after escalation → fall back to seeds.py + write `editorial_overrides`
